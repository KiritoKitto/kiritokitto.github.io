<script>
const grid = document.getElementById("track-grid");
const gameItems = games.filter(g => g.type === "Game").map(g => ({ ...g, scoreNum: g.score ? parseFloat(g.score) : 0 }));

function renderCards() {
    grid.innerHTML = '';
    const sezione1 = gameItems.filter(g => g.year === "2025" && ( g.scoreNum >= 8 || g.fav === "TRUE"));
    const sezione2 = gameItems.filter(g => g.year === "2025" && g.status === "Completed" && g.scoreNum <= 7 && g.fav !== "TRUE");
    const sezione3 = gameItems.filter(g => g.year === "2025" );
    const sezione4 = gameItems.filter(g => g.year === "2025" && g.status !== "Completed" && g.fav !== "TRUE");

    function renderSection(title, items) {
        if (!items.length) return;

        const section = document.createElement("div");
        section.className = "track-section";
        section.innerHTML = `<h2>${title}</h2>`;

        const sectionGrid = document.createElement("div");
        sectionGrid.className = "track-section-grid";

        items.forEach(g => {
        const card = document.createElement('div');
        card.className = 'trackcard';
        card.innerHTML = `
        <div class="image-wrapper">
        <img class="trackcard-image" src="${g.banner}" alt="${g.title}">
        </div>`;
        sectionGrid.appendChild(card);
        });

        section.appendChild(sectionGrid);
        const hr = document.createElement('hr');
        section.appendChild(hr);
        grid.appendChild(section);
    }

    renderSection("Best of 2025", sezione1.sort((a, b) => b.scoreNum - a.scoreNum));
    renderSection("Other Games", sezione2.sort((a, b) => b.scoreNum - a.scoreNum));
    renderSection("Dropped", sezione4.sort((a, b) => b.scoreNum - a.scoreNum));
}

renderCards();
</script>
